+import React from 'react';
+import { useCurrentFrame, useVideoConfig, spring } from 'remotion';
+import type { Segment } from '../CaptionEngine';
+import { loadFont } from '@remotion/google-fonts/Anton';
+
+const { fontFamily } = loadFont();
+
+export const MrBeastStyle: React.FC<{ segment: Segment | null }> = ({ segment }) => {
+  const frame = useCurrentFrame();
+  const { fps } = useVideoConfig();
+  const currentTime = frame / fps;
+
+  if (!segment) return null;
+
+  return (
+    <div style={{
+      position: 'absolute',
+      bottom: '25%',
+      width: '100%',
+      display: 'flex',
+      justifyContent: 'center',
+      alignItems: 'center',
+      flexWrap: 'wrap',
+      gap: '20px',
+      padding: '0 40px',
+    }}>
+      {segment.words.map((wordObj, i) => {
+        const isActive = currentTime >= wordObj.start && currentTime <= wordObj.end;
+        const isPast = currentTime >= wordObj.start;
+        if (!isPast && !isActive) return null;
+
+        const wordStartFrame = Math.floor(wordObj.start * fps);
+        const scale = spring({
+          fps,
+          frame: frame - wordStartFrame,
+          config: { damping: 8, mass: 0.6, stiffness: 300 }
+        });
+
+        const colors = ['#FFFFFF', '#00FFFF', '#FF00FF', '#FFFF00'];
+        const color = colors[i % colors.length];
+
+        return (
+          <span key={i} style={{
+            fontFamily,
+            fontSize: isActive ? '65px' : '45px',
+            fontWeight: 900,
+            textTransform: 'uppercase',
+            color: color,
+            WebkitTextStroke: '4px black',
+            textShadow: '8px 8px 0px rgba(0,0,0,1)',
+            transform: `scale(${scale}) rotate(${i % 2 === 0 ? 3 : -3}deg)`,
+            display: 'inline-block',
+            transition: 'font-size 0.1s ease',
+          }}>
+            {wordObj.word}
+          </span>
+        );
+      })}
+    </div>
+  );
+};
